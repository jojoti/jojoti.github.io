## 删除所有 none 镜像
docker rmi `docker images | grep  "<none>" | awk '{print $3}'`

## 删除现有容器未使用的所有镜像
docker image prune -a -y

## 停止所有没用使用的容器
docker ps -a | grep "Exited" | awk '{print $1 }'|xargs docker rm
